JSON 解析与序列化

JSON方法示例 
// JavaScript对象
let person = {
  name: "李四",
  age: 30,
  hobbies: ["音乐", "旅行"],
  address: {
    city: "上海",
    country: "中国"
  }
};

// 将JavaScript对象转换为JSON字符串（序列化）
let jsonString = JSON.stringify(person, null, 2);

// 将JSON字符串转换回JavaScript对象（解析）
let parsedPerson = JSON.parse(jsonString);

// 从JSON字符串中提取特定数据
let jsonData = '{"name":"王五","age":28,"email":"wangwu@example.com"}';
let user = JSON.parse(jsonData);

控制结构

控制结构示例 
// 条件语句
let score = 85;
let grade;

if (score >= 90) {
    grade = "优秀";
} else if (score >= 70) {
    grade = "良好";
} else {
    grade = "需要努力";
}

// for循环
let output = "";
for (let i = 1; i <= 5; i++) {
    output += "迭代 " + i + "\n";
}

// 数组迭代
let colors = ["红色", "绿色", "蓝色"];
let colorList = "";
colors.forEach(color => {
    colorList += "颜色: " + color + "\n";
});

// while循环
let count = 3;
let countdown = "";
while (count > 0) {
    countdown += count + "...";
    count--;
}
